sequenceDiagram
    actor RM as Restaurant Manager
    participant SYS as School Ordering System

    RM->>SYS: login
    SYS-->>RM: display main page

    RM->>SYS: open Manage Orders
    SYS-->>RM: display OrderList

    RM->>SYS: select order

    alt accept order

        SYS-->>RM: update order status

        SYS-->>RM: update order status
    end
